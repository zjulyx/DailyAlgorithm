```python
class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        res = 0
        mn = float('inf')
        for p in prices:
            mn = min(mn, p)
            res = max(res, p - mn)
        return res
```

#### C++

```cpp
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int res = 0;
        int minPrice = INT_MAX;

        for (auto p : prices) {
            minPrice = min(minPrice, p);
            res = max(res, p - minPrice);
        }

        return res;
    }
};
```
